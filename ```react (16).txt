```react
import React, { useState, useEffect, useRef } from 'react';
import { 
  Terminal, 
  Activity, 
  Trash2, 
  Play, 
  Database, 
  ShieldAlert, 
  Globe, 
  Clock, 
  FileJson,
  Server
} from 'lucide-react';

export default function App() {
  const [targetUrl, setTargetUrl] = useState('https://news.ycombinator.com/newest');
  const [isScraping, setIsScraping] = useState(false);
  const [results, setResults] = useState(null);
  const [logs, setLogs] = useState([]);
  const [systemState, setSystemState] = useState('IDLE'); // IDLE, ACTIVE, PURGED
  const [metrics, setMetrics] = useState({ latency: 0, itemsProcessed: 0, payloadTotal: 0 });
  
  const logEndRef = useRef(null);

  const addLog = (message, level = 'INFO') => {
    const timestamp = new Date().toISOString().split('T')[1].slice(0, 8);
    setLogs(prev => [...prev, `[${timestamp}] [${level}] ${message}`]);
  };

  useEffect(() => {
    if (logEndRef.current) {
      logEndRef.current.scrollIntoView({ behavior: 'smooth' });
    }
  }, [logs]);

  const handleScrape = async () => {
    if (!targetUrl) return;
    
    setIsScraping(true);
    setSystemState('ACTIVE');
    setResults(null);
    addLog(`Initiating extraction mission for target: ${targetUrl}`);
    addLog(`Resolving routing via CourlanRouter...`);
    
    // Simulate network latency and processing time
    const startMs = Date.now();
    
    setTimeout(() => {
      addLog(`FrontierManager: Target validated. Acquired lease.`);
      addLog(`GodEngine: Fetching payload...`);
    }, 600);

    setTimeout(() => {
      addLog(`HardenedDOMParser: Extracting nodes...`, 'DEBUG');
      addLog(`DataAlchemist: Normalizing payload structure...`);
      
      const endMs = Date.now();
      const latency = endMs - startMs;
      
      // Simulated response matching the G.O.D. API schema
      const mockResponse = {
        url: targetUrl,
        status: "SUCCESS",
        metrics: {
          payload_bytes: Math.floor(Math.random() * 50000) + 15000,
          discovered_anchors_count: Math.floor(Math.random() * 400) + 50,
          execution_ms: latency
        },
        extracted_data: {
          title: "New Links | Hacker News",
          body: "1. Show HN: A new secure pipeline\n2. The state of web scraping in 2026\n3. Ask HN: Postgres vs SQLite for local queues?",
          links: [
            "https://news.ycombinator.com/item?id=1",
            "https://news.ycombinator.com/item?id=2",
            "https://github.com/tangleroot013/god_stack"
          ]
        }
      };

      setResults(mockResponse);
      setMetrics(prev => ({
        latency: latency,
        itemsProcessed: prev.itemsProcessed + 1,
        payloadTotal: prev.payloadTotal + mockResponse.metrics.payload_bytes
      }));
      
      addLog(`Mission complete. Payload: ${mockResponse.metrics.payload_bytes} bytes.`);
      setIsScraping(false);
      setSystemState('IDLE');
    }, 1800);
  };

  const handleSecureWipe = () => {
    setSystemState('PURGING');
    
    // 1. Clear all in-memory UI state
    setTargetUrl('');
    setResults(null);
    setLogs([]);
    setMetrics({ latency: 0, itemsProcessed: 0, payloadTotal: 0 });
    
    // 2. Simulate the API call to the backend for footprint purging
    // fetch('/api/v1/system/purge_sensitive_footprint', { method: 'POST' })
    setTimeout(() => {
      setSystemState('PURGED');
    }, 800);
  };

  const handleRestore = () => {
    setSystemState('IDLE');
    addLog('System restored from purged state. Awaiting telemetry.');
  };

  if (systemState === 'PURGED') {
    return (
      <div className="min-h-screen bg-black flex flex-col items-center justify-center text-red-500 font-mono p-4">
        <ShieldAlert className="w-24 h-24 mb-6 animate-pulse" />
        <h1 className="text-3xl font-bold tracking-widest mb-2">SYSTEM PURGED</h1>
        <p className="text-gray-500 mb-8 max-w-md text-center">
          Memory state wiped. DOM cleared. Object URLs revoked. 
          Backend purge hook executed successfully.
        </p>
        <button 
          onClick={handleRestore}
          className="px-6 py-3 border border-red-900 text-red-700 hover:bg-red-950 hover:text-red-400 transition-colors rounded-sm"
        >
          REINITIALIZE CONSOLE
        </button>
      </div>
    );
  }

  return (
    <div className="min-h-screen bg-gray-950 text-gray-300 font-mono flex flex-col selection:bg-cyan-900 selection:text-cyan-50">
      
      {/* Top Navigation Bar */}
      <header className="border-b border-gray-800 bg-gray-900 px-6 py-3 flex items-center justify-between">
        <div className="flex items-center space-x-3">
          <Terminal className="w-5 h-5 text-cyan-500" />
          <h1 className="text-lg font-bold tracking-wider text-gray-100">G.O.D. STACK <span className="text-xs text-gray-500 font-normal ml-2">v2.2.0</span></h1>
        </div>
        <div className="flex items-center space-x-6">
          <div className="flex items-center space-x-2 text-xs">
            <span className={`w-2 h-2 rounded-full ${systemState === 'ACTIVE' ? 'bg-green-500 animate-pulse' : 'bg-gray-600'}`}></span>
            <span className="uppercase text-gray-400">{systemState}</span>
          </div>
          <button 
            onClick={handleSecureWipe}
            className="flex items-center space-x-2 px-3 py-1.5 bg-red-950 text-red-400 border border-red-900 rounded-sm hover:bg-red-900 hover:text-red-100 transition-colors text-sm"
            title="Wipe state and execute purge_sensitive_footprint()"
          >
            <Trash2 className="w-4 h-4" />
            <span>SECURE WIPE</span>
          </button>
        </div>
      </header>

      {/* Main Content Area */}
      <main className="flex-1 flex flex-col lg:flex-row p-4 gap-4 overflow-hidden">
        
        {/* Left Column: Controls & Logs */}
        <section className="flex flex-col w-full lg:w-1/3 space-y-4">
          
          {/* Input Panel */}
          <div className="bg-gray-900 border border-gray-800 rounded-sm p-4">
            <h2 className="text-sm font-semibold text-gray-400 mb-3 flex items-center">
              <Globe className="w-4 h-4 mr-2" />
              Target Acquisition
            </h2>
            <div className="flex flex-col space-y-3">
              <input 
                type="text" 
                value={targetUrl}
                onChange={(e) => setTargetUrl(e.target.value)}
                placeholder="https://..."
                className="w-full bg-black border border-gray-700 p-2 text-sm text-gray-200 focus:outline-none focus:border-cyan-500 transition-colors rounded-sm"
                disabled={isScraping || systemState === 'PURGING'}
              />
              <button 
                onClick={handleScrape}
                disabled={isScraping || systemState === 'PURGING' || !targetUrl}
                className="w-full flex items-center justify-center space-x-2 bg-cyan-900 text-cyan-100 p-2 hover:bg-cyan-800 disabled:opacity-50 disabled:cursor-not-allowed transition-colors rounded-sm border border-cyan-700"
              >
                {isScraping ? <Activity className="w-4 h-4 animate-spin" /> : <Play className="w-4 h-4" />}
                <span>{isScraping ? 'EXECUTING...' : 'EXECUTE MISSION'}</span>
              </button>
            </div>
          </div>

          {/* Telemetry Logs */}
          <div className="bg-gray-900 border border-gray-800 rounded-sm flex-1 flex flex-col min-h-[300px]">
            <div className="border-b border-gray-800 p-3 flex justify-between items-center">
              <h2 className="text-sm font-semibold text-gray-400 flex items-center">
                <Activity className="w-4 h-4 mr-2" />
                Live Telemetry
              </h2>
            </div>
            <div className="p-3 bg-black flex-1 overflow-y-auto text-xs space-y-1.5 custom-scrollbar">
              {logs.length === 0 ? (
                <span className="text-gray-700 italic">Awaiting instructions...</span>
              ) : (
                logs.map((log, i) => (
                  <div key={i} className={`${log.includes('[WARNING]') ? 'text-yellow-500' : log.includes('[DEBUG]') ? 'text-gray-500' : 'text-gray-300'}`}>
                    {log}
                  </div>
                ))
              )}
              <div ref={logEndRef} />
            </div>
          </div>
        </section>

        {/* Right Column: Results & Analytics */}
        <section className="flex flex-col w-full lg:w-2/3 space-y-4">
          
          {/* Metrics Overview */}
          <div className="grid grid-cols-3 gap-4">
            <div className="bg-gray-900 border border-gray-800 rounded-sm p-4 flex flex-col">
              <span className="text-gray-500 text-xs flex items-center mb-1"><Clock className="w-3 h-3 mr-1"/> P95 Latency</span>
              <span className="text-2xl text-cyan-400">{metrics.latency} <span className="text-sm text-gray-600">ms</span></span>
            </div>
            <div className="bg-gray-900 border border-gray-800 rounded-sm p-4 flex flex-col">
              <span className="text-gray-500 text-xs flex items-center mb-1"><Server className="w-3 h-3 mr-1"/> Items Processed</span>
              <span className="text-2xl text-purple-400">{metrics.itemsProcessed}</span>
            </div>
            <div className="bg-gray-900 border border-gray-800 rounded-sm p-4 flex flex-col">
              <span className="text-gray-500 text-xs flex items-center mb-1"><Database className="w-3 h-3 mr-1"/> Payload Volume</span>
              <span className="text-2xl text-emerald-400">{(metrics.payloadTotal / 1024).toFixed(2)} <span className="text-sm text-gray-600">KB</span></span>
            </div>
          </div>

          {/* Results Viewer */}
          <div className="bg-gray-900 border border-gray-800 rounded-sm flex-1 flex flex-col min-h-[400px]">
            <div className="border-b border-gray-800 p-3 flex justify-between items-center bg-gray-900">
              <h2 className="text-sm font-semibold text-gray-400 flex items-center">
                <FileJson className="w-4 h-4 mr-2" />
                Extracted Payload
              </h2>
              {results && (
                <span className="px-2 py-0.5 bg-emerald-950 text-emerald-400 border border-emerald-900 text-xs rounded-sm">
                  {results.status}
                </span>
              )}
            </div>
            
            <div className="flex-1 bg-[#0a0a0a] p-4 overflow-y-auto">
              {!results ? (
                <div className="h-full flex items-center justify-center text-gray-700">
                  No payload acquired
                </div>
              ) : (
                <div className="space-y-6">
                  {/* Metadata Block */}
                  <div className="grid grid-cols-2 gap-4 text-sm border-b border-gray-800 pb-4">
                    <div>
                      <span className="text-gray-500 block text-xs">Origin URL</span>
                      <a href={results.url} target="_blank" rel="noreferrer" className="text-cyan-500 hover:underline break-all">
                        {results.url}
                      </a>
                    </div>
                    <div>
                      <span className="text-gray-500 block text-xs">Discovered Anchors</span>
                      <span className="text-gray-300">{results.metrics.discovered_anchors_count}</span>
                    </div>
                  </div>
                  
                  {/* Data Content Block */}
                  <div>
                    <h3 className="text-gray-500 text-xs mb-2">Title</h3>
                    <p className="text-gray-200 font-sans">{results.extracted_data.title}</p>
                  </div>
                  
                  <div>
                    <h3 className="text-gray-500 text-xs mb-2">Extracted Body</h3>
                    <pre className="bg-black border border-gray-800 p-3 rounded-sm text-gray-400 whitespace-pre-wrap font-sans text-sm">
                      {results.extracted_data.body}
                    </pre>
                  </div>
                  
                  <div>
                    <h3 className="text-gray-500 text-xs mb-2">Sampled Links (First 3)</h3>
                    <ul className="space-y-1">
                      {results.extracted_data.links.map((link, idx) => (
                        <li key={idx} className="text-xs text-blue-400 hover:text-blue-300 cursor-pointer truncate">
                          {link}
                        </li>
                      ))}
                    </ul>
                  </div>
                </div>
              )}
            </div>
          </div>
        </section>
      </main>

      {/* Footer */}
      <footer className="border-t border-gray-800 bg-gray-950 p-2 text-center text-xs text-gray-600 flex justify-between px-6">
        <span>G.O.D. Orchestrator UI Component</span>
        <span>Environment: Production | Local Node</span>
      </footer>

      {systemState === 'PURGING' && (
        <div className="fixed inset-0 bg-red-950/90 flex items-center justify-center z-50 transition-opacity">
          <div className="text-red-500 flex flex-col items-center">
            <Trash2 className="w-16 h-16 mb-4 animate-bounce" />
            <h2 className="text-2xl font-bold tracking-widest">EXECUTING SECURE WIPE...</h2>
          </div>
        </div>
      )}

      <style dangerouslySetInnerHTML={{__html: `
        .custom-scrollbar::-webkit-scrollbar { width: 6px; }
        .custom-scrollbar::-webkit-scrollbar-track { background: #000; }
        .custom-scrollbar::-webkit-scrollbar-thumb { background: #333; border-radius: 3px; }
        .custom-scrollbar::-webkit-scrollbar-thumb:hover { background: #555; }
      `}} />
    </div>
  );
}
```
